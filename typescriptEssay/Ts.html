    # 类型守卫

    1.类型保护是可执行运行时检查的一种表达式，用于确保该类型在一定的范围内

2. 目前有 '4' 种方式来实现类型保护: 
  2.1 in 关键字
  2.2 typeof 关键字
  2.3 instanceof 关键字
  2.4 自定义类型保护的类型谓词('x is number')
    # 1.in 关键字
    interface zs {
  name: string;
  hobby: string[];
}

interface ls {
  name: string;
  sayHello(): vido;
}

type person = ls | zs;

function printEmployeeInformation(obj: person) {
  console.log("Name: " + obj.name); // 可以访问共同 属性 和 方法, 不会报错, 反之就会报错
  if ("hobby" in obj) { // 确保 'hobby' 在 obj 里面
    console.log("hobby: " + obj.hobby); // 为真所执行的 代码块
  }
  if ("sayHello" in obj) { // 确保 'sayHello' 在 obj 里面
    console.log("sayHello : " + obj.sayHello()); // 为真所执行的 代码块
  }
}

    # 1.函数重载示例：
    //重复定义了多次函数，前几次时函数的定义，最后一次是函数的实现
function diandao(x: number):number;
function diandao(x: string):string;
function diandao(x: number | string): number | string | void{
  if(typeof x === 'number'){
    return Number(x.toString().split('').reverse().join(''))
  }else if (typeof x === 'string'){
    return x.split('').reverse().join('')
  }
}
