})
.catch(() => {

})
.finally(() => {
    log('成功或者失败都会执行')

    # mixins混入

    1.是一个js对象

    2.包含script选中的任意功能选项 如：（data、components、methods、created、computed）

    3.将共用的功能以对象的方式传入mixins选项中，当组件使用mixins对象时所有mixins对象的选项都将被混入该组件本身的选项中来。

    4.提高代码的重用性，是代码干净和易于维护。

    5.多个组件中的数据或者功能很相近时，利用mixins将公共部分提取出来

    6.如何使用Mixins

    import { myMixins } from "@/Mixins/myMixins.js";
export default {
	mixins: [myMixins]
}

    7.注意

    Mixins.age = 18

    组件1：

    import { myMixins } from "@/Mixins/myMixins.js"
export default {
	mixins: [myMixins],
    data(){
        return{}
    },
    created(){this.age++}
}
//19

    组件2：

    export default {
  mixins: [myMixins],
  data() {
    return {}
  },
}
//18

    总结： 组件1改变的值，组件2初始值

    Mixins有合并冲突

    值为对象(components、methods 、computed、data)的选项，混入组件时选项会被合并，键冲突时优先组件，组件中的键会覆盖混入对象的

    值为函数(created、mounted)的选项，混入组件时选项会被合并调用，混合对象里的钩子函数在组件里的钩子函数之前调用

    # 类型守卫

    1.类型保护是可执行运行时检查的一种表达式，用于确保该类型在一定的范围内

2. 目前有 '4' 种方式来实现类型保护: 
  2.1 in 关键字
  2.2 typeof 关键字
  2.3 instanceof 关键字
  2.4 自定义类型保护的类型谓词('x is number')
    # 1.in 关键字
    interface zs {
  name: string;
  hobby: string[];
}

interface ls {
  name: string;
  sayHello(): vido;
}

type person = ls | zs;

function printEmployeeInformation(obj: person) {
  console.log("Name: " + obj.name); // 可以访问共同 属性 和 方法, 不会报错, 反之就会报错
  if ("hobby" in obj) { // 确保 'hobby' 在 obj 里面
    console.log("hobby: " + obj.hobby); // 为真所执行的 代码块
  }
  if ("sayHello" in obj) { // 确保 'sayHello' 在 obj 里面
    console.log("sayHello : " + obj.sayHello()); // 为真所执行的 代码块
  }
}

function diandao(x: number | string): number | string | void{
