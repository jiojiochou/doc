    # DOM

    • html元素在js中会被抽象成js对象, 它身上的attribute( 属性 )在js对象中叫property( 属性 )
    • 标准attribute可以在, js对象中以 . + 属性名的方式获取或者设置

    # 元素的获取

    • 如果我们获取到一个元素后, 可以根据这个元素去获取其他的元素, 我们称之为元素之间的导航;

    • 节点之间存在如下的关系

      • 父元素: parentElement;

        el.parentElement

      • 前兄弟节点: previousElementSibing;

        el.previousElementSibing

      • 后兄弟节点: nextElementSibing;

        el.nextElementSibing

      • 子节点: children;

        el.children // 类数组形式

      • 第一个子节点: firstElementChild;

        el.firstElementChild

      • 最后一个子节点: lastElementChild;

        el.lastElementChild

    # table 表格元素的获取

    • <table> 元素支持 ( 除了上面👆👆👆给出的, 之外 ) 以下属性也可以获取表格里面的元素:
      • table.rows --- <tr> 元素的集合;
      • table.caption/tHead/tFoot --- 引用元素 <caption> , <thead> , <tfoot> ;
      • table.tBodies --- <tbody> 元素的集合;
    • <thead> , <tfoot> , <tbody> 元素提供了 rows 属性:
      • tbody.rows --- 表格内部 <tr> 元素的集合;
    • <tr> :
      • tr.cells --- 在给定 <tr> 中的 <td> 和 <th> 单元格的集合;
      • tr.sectionRowIndex --- 给定的 <tr> 在封闭的 <thead> / <tbody> / <tfoot> 中的位置 ( 索引 );
      • tr.rowIndex --- 在整个表格中 <tr> 的编号 ( 包括表格的所有行 )

    # form 表单元素的获取

    • <form> 元素可以直接通过document来获取: document.forms;

      var formEl = document.forms[0] // 类数组

    • <form> 元素中的内容可以通过element来获取: form.elements;

      var elements = formEl.elements

    • 也可以通过 name 属性来获取表单的子元素;

        	<option>🍉🍉🍉</option>

        </section>

</body>

	var formEl = document.forms[0]
    var inputEl = formEl.elements.account // <input name='account'>
</script>

    # Event

    • 常见的事件列表
      • 鼠标事件:
        • click ----- 当鼠标点击一个元素的时 ( 触摸屏幕设备会在点击时生成 )

      • Document 事件:
        • DOMContentLoaded ----- 当 HTML 的加载和处理均完成, DOM 被完全构建完成时
      • CSS 事件:
        • transitionend ----- 当一个 CSS 动画完成时

    # 购物车

    <body>
  <table>
    <thead>
      <tr>
        <th>编号</th>
        <th>书籍名称</th>
        <th>出版日期</th>
        <th>价格</th>
        <th>购买数量</th>
        <th>操作</th>
      </tr>
    </thead>
    <tbody></tbody>
  </table>
  <h1>
    总价格: $ <span>0</span>
  </h1>
</body>
<script>
  let arrlist = [
    { id: 1, name: '<JavaScript>', date: '2006-01', price: 39.00, count: 5 },
    { id: 2, name: '<C++>', date: '2006-02', price: 59.00, count: 2 },
    { id: 3, name: '<MySQL>', date: '2006-03', price: 49.00, count: 1 },
    { id: 4, name: '<PHP>', date: '2006-04', price: 79.00, count: 10 },
  ]

  let tab = document.body.firstElementChild
  let tbodyEl = tab.tBodies[0]
  let priceEl = document.querySelector('span')
  for (let i = 0; i < arrlist.length; i++) {
    let trEl = document.createElement('tr')
    let item = Object.keys(arrlist[i])
    for (let j = 0; j < item.length; j++) {
      let tdEl = document.createElement('td')
      tdEl.textContent = arrlist[i][item[j]]
      trEl.append(tdEl)
    }
    let deltd = document.createElement('td')
    let delBtn = document.createElement('button')
    delBtn.textContent = '删除'
    deltd.append(delBtn)
    trEl.append(deltd)
    tbodyEl.append(trEl)

    delBtn.onclick = function () {
      // 删除对应行
      let parentRow = this.parentElement.parentElement
      let index = parentRow.sectionRowIndex
      parentRow.remove()
      // 删除对应数据
      arrlist.splice(index, 1)
      // 重新计算价格
      setPrice()
    }
  }

  function setPrice() {
    let price = 0
    for (let i = 0; i < arrlist.length; i++) {
      price += arrlist[i].price * arrlist[i].count
    }
    priceEl.textContent = price
  }
  setPrice()
</script>

console.log(location.host)

    • hostname :主机地址(不带端口);

      // 主机地址: 127.0.0.1 === localhost
console.log(location.hostname)

    • port :端口;

      // 只获取当前页面的端口: 8080
console.log(location.port)

    • pathname :路径;

      // 获取网址路径
console.log(location.pathname)

    • search :查询字符串;

      // 当前网页的查询字符串: ?name=bobodi&age=99
console.log(location.search)

    • hash :哈希值;

      // 获取当前网页的片段或者叫哈希, 不同的叫法但是指的同一个东西
console.log(location.hash)

    # URLSearchParams 构造函数

    用于处理URL的查询字符串

    • 可以将一个字符串转化为URLSearchParams类型;

    • 也可以将一个URLSearchParams类型转成字符串;

      var urlsearch = new URLSearchParams('?name=bobodi&age=18&height=1.66') // location.search
console.log(urlsearch.toString()) // name=bobodi&age=18&height1.88

    URLSearchParams常见方法

    • get :获取搜索参数的值

      var urlsearch = new URLSearchParams('?name=bobodi&age=18&height=1.66') // location.search
console.log(urlsearch.get('name')) // bobodi

    • set :设置一个搜索参数和值

      urlsearch.set('name','温签') // 修改
urlsearch.set('hobby','抽烟喝酒烫头') // 新增, 增加 参数

    • append :追加一个搜索参数和值

      urlsearch.append('weight',200) // 新增, 增加 参数

    • has :判断是否有某个搜索参数

      console.log(urlsearch.has('name')) // true

    # JSON

    # 在目前的开发中, JSON是一种非常重要的数据格式, 他并不是编程语言, 而是一种可以在服务器和客户端之间传输的数据格式
    # 其他的传输格式
    • XML :在早期的网络传输中主要是使用XML来进行数据交换的, 但是这种格式在解析, 传输等各方面都弱于JSON, 所以目前已经很少在被使用了;
    • Protobuf :另外一个在网络传输中目前已经越来越多使用的传输格式是protobuf, 但是直到2021年的3.x版本才支持JavaScript, 所以目前在前端使用的较少;

    # JSON基本语法

    • JSON的顶层支持三种类型的值
      • 简单值 :数字 ( Number ), 字符串 ( String, 不支持单引号 ), 布尔类型 ( Boolean ), null类型
      • 对象值 :由key, value组成, key是字符串类型, 并且必须添加双引号, 值可以是简单值, 对象值, 数组值
      • 数组值 :数组的值可以是简单值, 对象值, 数组值

    • parse() :得到相应的JavaScript 值
